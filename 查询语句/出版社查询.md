```plsql
-- 1.模糊查询出版社信息
SELECT * FROM PRESS WHERE PRENAME LIKE '%邮电%';

-- 2.查看一个出版社出的图书数量
SELECT Count(*)
FROM LIBRARY.BOOK_INFO
WHERE LIBRARY.BOOK_INFO.PRENAME = '人民邮电出版社';

-- 3.查看一个出版设出的所有图书
SELECT BOOKNO,BOOKNAME,ISBN,AUTNAME,PRENAME,PUBLICTIONTIME
FROM LIBRARY.BOOK_INFO
WHERE LIBRARY.BOOK_INFO.PRENAME = '人民邮电出版社';
```
